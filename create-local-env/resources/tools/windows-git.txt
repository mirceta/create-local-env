function gbranchtoclip {
	 $b = git branch | Select-String -Pattern '\*'
	 $b.ToString().Substring(2) | clip.exe
}

function gfulltocurrbranch {
	# side effect: clip changed

	Param(
        [Parameter(Mandatory=$true)][string]$message
    )

	$branch = gbranchtoclip
	$branch
	gfull $message $branch
}

function gfull {
	Param(
        [Parameter(Mandatory=$true)][string]$message,
		[Parameter(Mandatory=$false)][string]$branch
    )
	git status
	git add -A
	git commit -m $message
	if ($branch) {
		git push origin $branch
	}
}
